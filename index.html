검색 결과가 없습니다.

'; return; } data.item.forEach(book => { const item = document.createElement('div'); item.className = 'book-item'; item.innerHTML = `

`; item.onclick = () => window.open(book.link, '_blank'); resultsDiv.appendChild(item); }); }
